#include #include // ANSI Color Codes for attractive output #define RESET "\033[0m" #define BOLD "\033[1m" #define RED "\033[31m" #define GREEN "\033[32m" #define YELLOW "\033[33m" #define BLUE "\033[34m" #define CYAN "\033[36m" struct Student { int roll; char name[50]; int marks[5]; float total; float percent; char grade[30]; }; // Function to calculate total, percentage and grade void calculateResult(struct Student *s) { s->total = 0; for (int i = 0; i < 5; i++) { s->total += s->marks[i]; } s->percent = s->total / 5.0; if (s->percent > 75.0) { strcpy(s->grade, "DISTINGUISHED"); } else if (s->percent >= 60.0) { strcpy(s->grade, "FIRST DIVISION"); } else if (s->percent >= 50.0) { strcpy(s->grade, "SECOND DIVISION"); } else if (s->percent >= 40.0) { strcpy(s->grade, "THIRD DIVISION"); } else { strcpy(s->grade, "FAIL"); } } // Function to print nice banner void printHeader() { printf("%s", CYAN); printf("===========================================================\n"); printf(" STUDENT RESULT MANAGEMENT SYSTEM\n"); printf(" Using Array of Structures\n"); printf("===========================================================\n"); printf(RESET); } int main() { int n; printHeader(); printf("Enter number of students: "); scanf("%d", &n); if (n <= 0) { printf("%sError: Number of students must be positive!%s\n", RED, RESET); return 1; } struct Student students[100]; // Maximum 100 students char *subjectNames[5] = {"Mathematics", "Physics", "Chemistry", "English", "Computer Science"}; // Input details for all students for (int i = 0; i < n; i++) { printf("\n%s--- Enter details for Student %d ---%s\n", BOLD, i+1, RESET); printf("Roll Number: "); scanf("%d", &students[i].roll); printf("Name: "); getchar(); // to consume newline fgets(students[i].name, 50, stdin); students[i].name[strcspn(students[i].name, "\n")] = '\0'; for (int j = 0; j < 5; j++) { do { printf("Marks in %s (0-100): ", subjectNames[j]); scanf("%d", &students[i].marks[j]); if (students[i].marks[j] < 0 || students[i].marks[j] > 100) { printf("%sMarks should be between 0 and 100!%s\n", RED, RESET); } } while (students[i].marks[j] < 0 || students[i].marks[j] > 100); } calculateResult(&students[i]); } // Bubble sort to find merit list (highest percentage first) for (int i = 0; i < n - 1; i++) { for (int j = 0; j < n - i - 1; j++) { if (students[j].percent < students[j + 1].percent) { struct Student temp = students[j]; students[j] = students[j + 1]; students[j + 1] = temp; } } } // Display result table printf("\n%s===========================================================\n", GREEN); printf(" FINAL MERIT LIST\n"); printf("===========================================================%s\n", RESET); printf("%-4s %-6s %-22s %-10s %-6s %-6s %-6s %-6s %-7s %-8s %-15s\n", "Rank", "Roll", "Name", "Math", "Phy", "Chem", "Eng", "CS", "Total", "Percent", "Grade"); printf("--------------------------------------------------------------------------------------------\n"); for (int i = 0; i < n; i++) { char *colorCode = RESET; if (strcmp(students[i].grade, "DISTINGUISHED") == 0) colorCode = GREEN; else if (strcmp(students[i].grade, "FIRST DIVISION") == 0) colorCode = BLUE; else if (strcmp(students[i].grade, "SECOND DIVISION") == 0) colorCode = YELLOW; else if (strcmp(students[i].grade, "THIRD DIVISION") == 0) colorCode = CYAN; else colorCode = RED; printf("%-4d %-6d %-22s %-10d %-6d %-6d %-6d %-6d %-7.0f %-8.2f %s%-15s%s\n", i+1, students[i].roll, students[i].name, students[i].marks[0], students[i].marks[1], students[i].marks[2], students[i].marks[3], students[i].marks[4], students[i].total, students[i].percent, colorCode, students[i].grade, RESET); } // Class Topper Congratulations if (n > 0) { printf("\n%s===========================================================\n", YELLOW); printf(" 🏆 CLASS TOPPER 🏆\n"); printf("===========================================================%s\n", RESET); printf("%sCongratulations %s!%s\n", BOLD, students[0].name, RESET); printf("Roll Number : %d\n", students[0].roll); printf("Percentage : %.2f%%\n", students[0].percent); printf("Grade : %s%s%s\n", GREEN, students[0].grade, RESET); if (students[0].percent > 90.0) printf("%sOutstanding performance! You made the class proud!%s\n", GREEN, RESET); else if (students[0].percent > 75.0) printf("%sExcellent work! Keep it up!%s\n", GREEN, RESET); } // Simple class summary float classTotal = 0; for (int i = 0; i < n; i++) { classTotal += students[i].percent; } printf("\nClass Average Percentage : %.2f%%\n", classTotal / n); // Save result to file FILE *file = fopen("student_result.txt", "w"); if (file != NULL) { fprintf(file, "STUDENT RESULT - MERIT LIST\n\n"); for (int i = 0; i < n; i++) { fprintf(file, "Rank: %d Roll: %d Name: %s Total: %.0f Percent: %.2f Grade: %s\n", i+1, students[i].roll, students[i].name, students[i].total, students[i].percent, students[i].grade); } fclose(file); printf("\nResult has been saved to student_result.txt\n"); } printf("\n%sProgram completed successfully!%s\n", BOLD, RESET); return 0; }
